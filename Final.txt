-- Step 1: See duplicates first (run this to verify)
SELECT case_id, COUNT(*) as count
FROM public.derivatives_operations
GROUP BY case_id
HAVING COUNT(*) > 1
ORDER BY count DESC;

-- Step 2: Delete duplicates — keeps the LATEST row (highest id)
-- Safe to run — only deletes older duplicate rows
DELETE FROM public.derivatives_operations
WHERE id NOT IN (
  SELECT MAX(id)
  FROM public.derivatives_operations
  GROUP BY case_id
);

-- Step 3: Verify — should return empty now
SELECT case_id, COUNT(*) as count
FROM public.derivatives_operations
GROUP BY case_id
HAVING COUNT(*) > 1;
